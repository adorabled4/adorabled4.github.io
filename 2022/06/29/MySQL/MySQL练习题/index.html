4、不准用组函数（Max），取得最高薪水（给出两种解决方案）

mysql>  select ename ,sal from emp order by sal asc limit 13,1;
+-------+---------+
| ename | sal     |
+-------+---------+
| KING  | 5000.00 |
+-------+---------+
1 row in set (0.00 sec)

//表的自连接
mysql> select sal from emp where sal not in(select emp.sal from emp join (select sal from emp) emp2 on emp.sal >emp2.sal);
+--------+
| sal    |
+--------+
| 900.00 |
+--------+
1 row in set (0.00 sec)

mysql>  select sal from emp where sal 
not in
	(select 
     	emp.sal 
     from 	
     	emp 	
     join (select sal from emp) emp2 
     on 	emp.sal <emp2.sal);
+---------+
| sal     |
+---------+
| 5000.00 |
+---------+
1 row in set (0.00 sec)

3的原理
select  e1.sal from emp e1 join( select sal from emp )e2 on
e1.sal<e2.sal ; 没有大于5000的 ，因此当 e1.sal =5000(最大薪资) 时，不符合on的条件，所以该跳查询语句的结果没有5000 ，配合not in 就可以得出最大薪资

2.链接dept表， 获取dname
select   e.ename ,d.dname ,e.deptno

		(select ename, deptno from emp where job='CLERK' ) e

4.通过deptno 连接两张表 （这三张表都是通过deptno 连接的）  
select t.* ,d.cnt from 
        (select   e.ename ,d.dname ,e.deptno
        from dept d 
        join (select ename, deptno from emp where job='CLERK' ) e
        on   e.deptno=d.deptno
        ) t
join (select deptno,count(ename) as cnt from emp group by deptno) d
on d.deptno =t.deptno

select 
	t1.*,t2.deptcount
from
	(select 
		e.ename,e.job,d.dname,d.deptno
	from 
		emp e
	join
		dept d
	on
		e.deptno = d.deptno
	where 
		e.job = 'CLERK') t1
join
	(select deptno, count(*) as deptcount from emp group by deptno) t2
on
	t1.deptno = t2.deptno;

d.deptno, count(e.ename) as cnt,ifnull( avg(sal),0) as avgsal,ifnull(avg(timestampdiff(DAY, hiredate, now())), 0) as avgservicetime_DAY

34、给任职日期超过30年的员工加薪10%

TimeStampDiff

select ename from emp where  TimeStampDiff(YEAR,HIREDATE,now()) >=30 

update emp set sal=sal*1.1 where TimeStampDiff(YEAR,HIREDATE,now()) >=30
